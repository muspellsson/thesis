\begin{frame}
\frametitle{Термодинамика}
\begin{itemize}
\item Феноменологическая наука
\item Изучает поведение макросистем
\item Близка к статистической физике
\item Рассматривает явления как процессы преобразования энергии
\item Позволяет рассматривать макросистемы как объекты управления
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Основные понятия термодинамики}
\begin{itemize}
\item Координата и потенциал
\begin{itemize}
\item Изменение координаты свидетельствует о протекании процесса
\item Разница потенциалов является движущей силой
\end{itemize}
\item Экстенсивные и интенсивные переменные
\begin{itemize}
\item Интенсивные переменные не изменяются при объединении двух идентичных подсистем
\item Экстенсивные переменные аддитивны
\end{itemize}
\item Равновесие
\begin{itemize}
\item Наступает если величины потенциалов каждого рода по объему системы равны
\item После протекания процесса система возвращается в состояние равновесия: релаксация
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Равновесные и неравновесные процессы}
\begin{itemize}
\item Классическая термодинамика
\begin{itemize}
\item Изучает равновесные процессы
\item Равновесные процессы могут быть обратимы
\end{itemize}
\item Неравновесная термодинамика
\begin{itemize}
\item Изучает неравновесные необратимые процессы
\item Разбивает систему на бесконечно малые равновесные элементы
\end{itemize}
\item Оптимизационная термодинамика
\begin{itemize}
\item Изучает неравновесные необратимые процессы
\item Разбивает систему на конечное число равновесных подсистем
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимизационная термодинамика процессов}
\begin{itemize}
\item Балансовые уравнения
\begin{itemize}
\item Материальный баланс
\item Энергетический баланс
\item Энтропийный баланс
\end{itemize}
\item В энтропийный баланс входит неотрицательное слагаемое $\sigma$ --- диссипация
\item В реальности $\sigma \neq 0$
\item Условие $\sigma \geq \sigma_{min}$ определяет область реализуемости
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Два типа процессов}
\begin{itemize}
\item Производительность системы зависит от потока вырабатываемой механической, электрической мощности или мощности разделения, поток затрат представлен потоком тепла
\begin{itemize}
\item Область реализуемости ограничена сверху
\end{itemize}
\item Производительность определяется потоком теплоты, а затраты --- механической или электрической мощностью или затратами тепла
\begin{itemize}
\item Производительность монотонно возрастает с ростом затрат
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Оптимизационная термодинамика систем}
\begin{itemize}
\item Реальные процессы многостадийны
\item Оптимизационная термодинамика систем из нескольких аппаратов до сих пор не была разработана
\item Множество технологических параметров системы отображается в меньшее множество характеристических коэффициентов
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Бинарная ректификация}
\begin{itemize}
\item Граница области реализуемости без учета необратимости
\[
g = bq
\]
\item Граница области реализуемости с учетом необратимости тепло- и массопереноса
\[
g = bq-aq^2,
\]
\item $g$, $q$ --- потоки
\item $b$, $a$ --- характеристические коэффициенты
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Тернарная ректификация}
\begin{itemize}
\item Разделение на каскаде из двух колонн
\item Два варианта разделения
\item $b_{ij}$ --- характеристические коэффициенты колонн
\item Условия полной термодинамической согласованности
\[
\frac{b_{11}^2(1-x_0)}{a_{11}} = \frac{b_{12}^2}{a_{12}},
\]
\[
\frac{b_{21}^2(1-x_2)}{a_{21}} = \frac{b_{21}^2}{a_{22}}
\]
\item Граница области реализуемости для одного из вариантов
\[
g = \frac{b_{11}b_{12}}{b_{12}+b_{11}(1-x_0)}(q_1 + q_2) - \frac{a_{11}b_{12}}{b_{12} + b_{11}(1-x_0)}(q_1 + q_2)^2
\]
\item Параметризуется как
\[
g = b^I q - a^I q^2
\]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Многокомпонентная ректификация}
\begin{itemize}
\item Понятие о квазикомпонентах
\item Можно представить как комбинацию бинарных и тернарных ректификаций
\item Граница области реализуемости параметризуется как
\[
g = b q - a q^2
\]
\item Можно аналитически найти оптимальный порядок разделения
\item При невыполнении условий согласованности --- численное решение
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Нетермическое разделение}
\begin{itemize}
\item Мембраны и центрифуги
\item Обратимая оценка мощности разделения
	\[
	p^0 = -R g_0 T\sum\limits_{i=1}^n{x_i\ln{x_i}} = b g_0.
	\]
\item Необратимые затраты мощности разделения
\[
\Delta p_j = \frac{g^2_0 T}{S_j}\left[\left(\sum\limits_{i=1}^{\nu}x_i\right)^2d_{\nu+} + \left(\sum\limits_{i=\nu+1}^{n}d_{(\nu+1)-}\right)\right]
\]
\item Итоговая мощность разделения
\[
p = p^0 - \Delta p = b g_0 - a g_0^2
\]
\item Задача об оптимальном распределении поверхностей контакта
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Многопоточный теплообмен}
\begin{itemize}
\item Суммарный коэффициент теплообмена $\overline{\alpha}$, суммарная тепловая нагрузка $\overline{q}$
\item Температуры греющих потоков  на входе $T_{i0}$, водяные эквиваленты $W_i$
\[
\overline{\sigma}_{min} = \overline{\alpha}\frac{(1-m)^2}{m},
\]
где

\[
\left\{
\begin{array}{l}
m=1-\frac{1}{\overline{\alpha}}\sum\limits_{i=1}^k W_i (\ln T_{i0}- \ln \overline{T}),\\
\overline{T}= \frac{\sum\limits_{i=1}^k T_{i0} W_i - \overline{q}}{\sum\limits_{i=1}^k W_i}.
\end{array}
\right.
\]
\item Условие термодинамической реализуемости
\[
\sum\limits_{\nu=1}^M W_{\nu}\ln\frac{T_{\nu}\mbox{вх}}{T_{\nu}\mbox{вых}}\geq\overline{\sigma}_{min}.
\]
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Результаты}
\begin{itemize}
\item Множество технологических параметров многокомпонентной системы может быть сведено к меньшему множеству характеристических коэффициентов
\item Одиночные аппараты разделены на два типа в зависимости от вида границы области реализуемости
\item Одиночные аппараты могут произвольным образом комбинироваться
\item Результаты могут быть использованы для создания общей теории оптимизационной термодинамики многоаппаратных систем
\end{itemize}
\end{frame}
